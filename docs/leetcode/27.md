```py
class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        length = len(nums)
        k = length
        index = []
        for ind in range(length):
            if nums[ind] == val:
                index.append(ind)
                k -= 1
        for ind in sorted(index, reverse=True):
            nums.pop(ind)
        return k
```
