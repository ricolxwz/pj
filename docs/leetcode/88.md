func merge(nums1 []int, m int, nums2 []int, n int)  {
    f1 := 0
    f2 := 0
    temp := []int{}
    for f1 < m && f2 < n {
        if nums1[f1] <= nums2[f2] {
            temp = append(temp, nums1[f1])
            f1 += 1
        } else {
            temp = append(temp, nums2[f2])
            f2 += 1
        }
    }
    for f1 < m {
        temp = append(temp, nums1[f1])
        f1 += 1
    }
    for f2 < n {
        temp = append(temp, nums2[f2])
        f2 += 1
    }
    copy(nums1, temp)
}
